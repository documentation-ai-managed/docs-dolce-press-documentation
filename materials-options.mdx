// Material options object
const materials = {
  cotton: { weight: '100lb', finish: 'soft gloss', price: 0.15 },
  linen: { weight: '80lb', finish: 'linen', price: 0.12 },
  recycled: { weight: '90lb', finish: 'matte', price: 0.10 }
};

// Calculate total cost
function calculateTotal(quantity, material, printing) {
  return quantity * (materials[material].price + printing.cost);
}

  ```python
# Material compatibility check
def check_compatibility(material, printing_method):
    compatibilities = {
        'cotton': ['digital', 'letterpress'],
        'linen': ['letterpress', 'foil'],
        'recycled': ['digital', 'offset']
    }
    return printing_method in compatibilities.get(material, [])
  ```
